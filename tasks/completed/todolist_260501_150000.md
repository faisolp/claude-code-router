```json
{

    {
      "name": "glm",
      "api_base_url": "https://open.bigmodel.cn/api/paas/v4/chat/completions",
      "api_key": "your-glm-api-key",
      "models": ["glm-4", "glm-4-plus", "glm-4-flash"],

        "use": ["Anthropic"]
      }

  ]
}
```
